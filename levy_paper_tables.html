```{r}
#| label: setup
#| include: false

library(htmltools)
library(fs)

# Tables are created by scripts/descriptive_stats_22_lags_cluster_updated.R
# after running the model script locally.
table_dir <- fs::path("results", "tables")

include_table <- function(file) {
  table_path <- fs::path(table_dir, file)

  if (!fs::file_exists(table_path)) {
    htmltools::HTML(
      paste0(
        "<p><strong>Missing table:</strong> <code>",
        htmltools::htmlEscape(as.character(table_path)),
        "</code></p>"
      )
    )
  } else {
    htmltools::includeHTML(table_path)
  }
}
```
